git branch  branch_name	                创建本地分支

git checkout branch_name	              切换本地分支

git branch -d branch_name               删除本地分支

git branch -D branch_name               强制删除本地分支

git push origin --delete branch_name    删除远程仓库分支

git branch -m old_branch new_branch     修改本地分支名

git push origin :old_branch             删除旧的远程仓库名

git push --set-upstream origin new_branch 创建新的远程仓库名

git branch -a：查看本地和远程仓库的所有分支

git branch -r：查看所有远程分支
